import { useState } from "react";

export default function Pricing() {
  const [loading, setLoading] = useState(false);

  async function handleCheckout(type) {
    setLoading(true);
    const endpointMap = {
      standard: "/api/create-standard-session",
      pro: "/api/create-pro-session",
      lifetime: "/api/create-lifetime-session",
    };
    const res = await fetch(endpointMap[type], { method: "POST" });
    const { url } = await res.json();
    window.location.href = url;
  }

  return (
    <div className="pricing text-center py-12 bg-gray-50 min-h-screen">
      <h1 className="text-4xl font-bold mb-8">Access Delegate Lens</h1>
      <div className="flex flex-wrap justify-center gap-8">
        <div className="bg-white shadow-md rounded-xl p-6 w-72">
          <h2 className="text-2xl font-semibold mb-2">Standard</h2>
          <p className="text-gray-500 mb-4">Essential executive dashboard</p>
          <p className="text-3xl font-bold mb-6">$45<span className="text-base font-normal">/mo</span></p>
          <button
            onClick={() => handleCheckout("standard")}
            disabled={loading}
            className="bg-black text-white px-6 py-3 rounded-lg w-full"
          >
            Start Standard
          </button>
        </div>

        <div className="bg-white shadow-lg border-2 border-indigo-600 rounded-xl p-6 w-72">
          <h2 className="text-2xl font-semibold mb-2 text-indigo-600">Pro</h2>
          <p className="text-gray-500 mb-4">Priority access + analytics suite</p>
          <p className="text-3xl font-bold mb-6">$89<span className="text-base font-normal">/mo</span></p>
          <button
            onClick={() => handleCheckout("pro")}
            disabled={loading}
            className="bg-indigo-600 text-white px-6 py-3 rounded-lg w-full"
          >
            Upgrade to Pro
          </button>
        </div>

        <div className="bg-white shadow-md rounded-xl p-6 w-72">
          <h2 className="text-2xl font-semibold mb-2">Lifetime</h2>
          <p className="text-gray-500 mb-4">Lifetime access, one-time payment</p>
          <p className="text-3xl font-bold mb-6">$445<span className="text-base font-normal"> one-time</span></p>
          <button
            onClick={() => handleCheckout("lifetime")}
            disabled={loading}
            className="bg-green-600 text-white px-6 py-3 rounded-lg w-full"
          >
            Buy Lifetime
          </button>
        </div>
      </div>
    </div>
  );
}
